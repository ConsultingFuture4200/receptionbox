    ```python
    """AUDIT-01 (co-residency stack-load) + AUDIT-03 (engine-swap-under-load).

    5-minute sustained N=2 concurrent E2E load with all 3 model classes loaded.
    Mid-run (at T+2:30) flips tts.primary in config/sanity_strata.yaml; measures
    swap time as delta between row write and first audio chunk from new engine.
    """
    from __future__ import annotations
    import asyncio
    import datetime
    import io
    import json
    import logging
    import os
    import pathlib
    import random
    import subprocess
    import time

    import yaml

    logger = logging.getLogger(__name__)

    POLL_INTERVAL_S = 10
    DEFAULT_DURATION_S = 300
    DEFAULT_CONCURRENCY = 2
    SWAP_AT_S = 150  # T+2:30 per D-Discretion
    SANITY_STRATA = pathlib.Path("config/sanity_strata.yaml")
    AUDIT_DIR = pathlib.Path("audit")
    RESULTS_DIR = pathlib.Path("results/audit_01")


    async def _poll_rocm_smi(stop_evt: asyncio.Event, log_path: pathlib.Path) -> None:
        while not stop_evt.is_set():
            try:
                proc = await asyncio.create_subprocess_exec(
                    "rocm-smi", "--showmeminfo", "vram", "--showuse", "--json",
                    stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE,
                )
                stdout, _ = await proc.communicate()
                data = json.loads(stdout.decode())
                for card_key, card in (data or {}).items():
                    if not card_key.startswith("card"):
                        continue
                    total_b = int(card.get("VRAM Total Memory (B)", 0))
                    used_b = int(card.get("VRAM Total Used Memory (B)", 0))
                    util = card.get("GPU use (%)", "0")
                    pct = (100.0 * used_b / total_b) if total_b else 0.0
                    row = {"ts": time.time(), "card": card_key, "vram_pct": pct,
                           "gpu_util": util, "vram_used_gb": used_b / (1024**3),
                           "vram_total_gb": total_b / (1024**3)}
                    with open(log_path, "a") as f:
                        f.write(json.dumps(row) + "\n")
                    if pct > 90.0:
                        logger.warning(f"[audit-01] VRAM pressure: {pct:.1f}% on {card_key}")
            except Exception as e:
                logger.warning(f"[audit-01] rocm-smi probe failed: {type(e).__name__}: {e}")
            try:
                await asyncio.wait_for(stop_evt.wait(), timeout=POLL_INTERVAL_S)
            except asyncio.TimeoutError:
                continue


    def _check_dmesg_errors() -> list[str]:
        """Return list of amdgpu errors from recent dmesg output."""
        try:
            r = subprocess.run(["dmesg", "--since", "30s", "--no-pager"],
                            capture_output=True, text=True, timeout=10)
            return [ln for ln in (r.stdout or "").splitlines() if "amdgpu" in ln.lower() and ("error" in ln.lower() or "fault" in ln.lower())]
        except Exception:
            return []


    async def _run_e2e_call(substrate, audio_bytes: bytes, prompt: str) -> tuple[bool, float]:
        t0 = time.monotonic()
        try:
            async def _audio_gen():
                yield audio_bytes
            stt_chunks = []
            async for c in substrate.transcribe(_audio_gen(), sample_rate=16000):
                stt_chunks.append(c)
            hyp = " ".join(c.text for c in stt_chunks) or "(empty STT)"
            llm_chunks = []
            async for c in substrate.generate(prompt + " " + hyp, max_tokens=32):
                llm_chunks.append(c.text)
            text = "".join(llm_chunks) or "okay"
            buf = bytearray()
            async for b in substrate.synthesize(text):
                buf.extend(b)
            return (True, (time.monotonic() - t0) * 1000.0)
        except Exception as e:
            logger.warning(f"[audit-01] e2e call failed: {type(e).__name__}: {e}")
            return (False, (time.monotonic() - t0) * 1000.0)


    def _flip_tts_primary(target: str) -> float:
        """Atomically rewrite config/sanity_strata.yaml; return write timestamp."""
        data = yaml.safe_load(SANITY_STRATA.read_text()) or {}
        data.setdefault("tts", {})["primary"] = target
        tmp = SANITY_STRATA.with_suffix(".yaml.tmp")
        tmp.write_text(yaml.safe_dump(data, sort_keys=False))
        ts = time.time()
        os.replace(tmp, SANITY_STRATA)
        logger.info(f"[audit-03] flipped tts.primary to {target} at ts={ts}")
        return ts


    async def _measure_swap_time(initial_primary: str, substrate, prompt: str) -> tuple[float | None, float]:
        """Returns (swap_time_ms_or_None, first_audio_chunk_ts).

        Calls synthesize() in a tight loop. The first call after the config write
        will trigger _read_tts_primary() re-read inside the substrate.
        """
        target = "kokoro" if initial_primary == "chatterbox" else "chatterbox"
        write_ts = _flip_tts_primary(target)
        first_chunk_ts: float | None = None
        try:
            async for b in substrate.synthesize(prompt + " (post-swap)"):
                if first_chunk_ts is None and b:
                    first_chunk_ts = time.time()
                    break
        except Exception as e:
            logger.warning(f"[audit-03] post-swap synthesize failed: {e}")
            return (None, write_ts)
        if first_chunk_ts is None:
            return (None, write_ts)
        return ((first_chunk_ts - write_ts) * 1000.0, first_chunk_ts)


    async def run_audit(*, duration_s: int = DEFAULT_DURATION_S,
                        concurrency: int = DEFAULT_CONCURRENCY,
                        swap_at_s: int = SWAP_AT_S,
                        run_id: str | None = None) -> dict:
        from substrate.rocm import ROCmSubstrate

        run_id = run_id or datetime.datetime.utcnow().strftime("audit_co_residency_%Y%m%dT%H%M%SZ")
        AUDIT_DIR.mkdir(parents=True, exist_ok=True)
        RESULTS_DIR.mkdir(parents=True, exist_ok=True)
        smi_log = RESULTS_DIR / "rocm_smi.jsonl"
        smi_log.write_text("")  # truncate

        substrate = ROCmSubstrate(
            vllm_url=os.environ.get("VLLM_URL", "http://localhost:8000"),
            vllm_model=os.environ.get("VLLM_MODEL", "/models/qwen3_4b_awq"),
            whisper_model_dir=os.environ.get("WHISPER_DIR", "/models/whisper"),
            chatterbox_url=os.environ.get("CHATTERBOX_URL", "http://localhost:8004"),
            kokoro_url=os.environ.get("KOKORO_URL", "http://localhost:8005"),
        )
        await substrate.load_stt()
        await substrate.load_llm()
        await substrate.load_tts()

        initial_primary = substrate._read_tts_primary()
        logger.info(f"[audit-01] initial tts.primary={initial_primary}")

        stop_evt = asyncio.Event()
        poll_task = asyncio.create_task(_poll_rocm_smi(stop_evt, smi_log))

        # Sustained N=concurrency E2E loop
        deadline = time.monotonic() + duration_s
        swap_done_at: float | None = None
        swap_time_ms: float | None = None
        swap_first_chunk_ts: float | None = None
        kernel_errors: list[str] = []
        call_results: list[tuple[bool, float]] = []
        # synthetic audio bytes (~1 sec of silence at 16k mono) for substrate.transcribe
        synth_audio = b"\x00\x00" * 16000
        synth_prompt = "Reply briefly."

        try:
            tick = time.monotonic()
            while time.monotonic() < deadline:
                # Engine swap at T+swap_at_s (AUDIT-03)
                elapsed = time.monotonic() - (deadline - duration_s)
                if swap_done_at is None and elapsed >= swap_at_s:
                    swap_time_ms, swap_first_chunk_ts = await _measure_swap_time(initial_primary, substrate, synth_prompt)
                    swap_done_at = time.monotonic()
                    logger.info(f"[audit-03] swap_time_ms={swap_time_ms}")

                # Run N=concurrency e2e calls in parallel
                batch = await asyncio.gather(*[
                    _run_e2e_call(substrate, synth_audio, synth_prompt)
                    for _ in range(concurrency)
                ], return_exceptions=False)
                call_results.extend(batch)

                # Every 30s check dmesg for amdgpu errors
                if (time.monotonic() - tick) > 30:
                    kernel_errors.extend(_check_dmesg_errors())
                    tick = time.monotonic()
                    if kernel_errors:
                        logger.warning(f"[audit-01] kernel errors detected; aborting run: {kernel_errors[:3]}")
                        break
        finally:
            stop_evt.set()
            await poll_task
            # Restore initial tts.primary so downstream gates aren't affected
            _flip_tts_primary(initial_primary)

        # Summarize VRAM trajectory from smi_log
        vram_points = [json.loads(l) for l in smi_log.read_text().splitlines() if l]
        max_vram_pct = max((p["vram_pct"] for p in vram_points), default=0.0)
        ok_calls = sum(1 for ok, _ in call_results if ok)

        audit_01_pass = (max_vram_pct < 90.0
                        and not kernel_errors
                        and ok_calls >= int(0.9 * len(call_results)))
        audit_03_pass = (swap_time_ms is not None and swap_time_ms < 30000)

        summary = {
            "run_id": run_id,
            "duration_s": duration_s,
            "concurrency": concurrency,
            "initial_primary": initial_primary,
            "total_calls": len(call_results),
            "ok_calls": ok_calls,
            "fail_calls": len(call_results) - ok_calls,
            "max_vram_pct": max_vram_pct,
            "kernel_errors": kernel_errors,
            "swap_time_ms": swap_time_ms,
            "swap_target": "kokoro" if initial_primary == "chatterbox" else "chatterbox",
            "audit_01_pass": audit_01_pass,
            "audit_03_pass": audit_03_pass,
            "timestamp_utc": datetime.datetime.utcnow().isoformat(),
        }
        (RESULTS_DIR / "run.json").write_text(json.dumps(summary, indent=2))
        write_audit_md(summary, vram_points, AUDIT_DIR)
        return summary


    def write_audit_md(summary: dict, vram_points: list[dict], audit_dir: pathlib.Path) -> None:
        # Try to embed a tiny matplotlib chart of VRAM trajectory
        plot_path = audit_dir / "co_residency_vram.svg"
        try:
            import matplotlib
            matplotlib.use("Agg")
            import matplotlib.pyplot as plt
            ts0 = vram_points[0]["ts"] if vram_points else 0
            xs = [(p["ts"] - ts0) for p in vram_points]
            ys = [p["vram_pct"] for p in vram_points]
            fig, ax = plt.subplots(figsize=(8, 3))
            ax.plot(xs, ys, "-o", markersize=3)
            ax.axhline(90, linestyle="--", color="red", label="OOM threshold 90%")
            ax.set_xlabel("seconds")
            ax.set_ylabel("VRAM %")
            ax.set_title(f"AUDIT-01 VRAM trajectory (max={summary['max_vram_pct']:.1f}%)")
            ax.legend(loc="best")
            fig.tight_layout()
            fig.savefig(plot_path)
            plt.close(fig)
        except Exception as e:
            logger.warning(f"[audit] plot failed: {e}")

        md_path = audit_dir / "co_residency_run.md"
        md = [
            "# Co-Residency + Engine-Swap Audit (AUDIT-01 + AUDIT-03)",
            "",
            f"**Run ID:** {summary['run_id']}",
            f"**Timestamp:** {summary['timestamp_utc']}",
            f"**Duration:** {summary['duration_s']} s",
            f"**Concurrency:** N={summary['concurrency']}",
            "",
            f"## AUDIT-01: {'PASS' if summary['audit_01_pass'] else 'FAIL'}",
            "",
            f"- Max VRAM: **{summary['max_vram_pct']:.2f}%** (threshold: 90%)",
            f"- E2E calls: {summary['ok_calls']}/{summary['total_calls']} OK",
            f"- Kernel errors detected: {len(summary['kernel_errors'])}",
            "",
            f"![VRAM trajectory](co_residency_vram.svg)",
            "",
            f"## AUDIT-03: {'PASS' if summary['audit_03_pass'] else 'FAIL'}",
            "",
            f"- Swap source: `{summary['initial_primary']}` → target: `{summary['swap_target']}`",
            f"- Swap time: **{summary['swap_time_ms']:.0f} ms** (threshold: 30000 ms)" if summary['swap_time_ms'] else "- Swap time: NOT MEASURED (failed)",
            "",
            "## Failure Modes (if any)",
            "",
            *(["- Kernel errors:"] + [f"  - {ln}" for ln in summary["kernel_errors"][:5]] if summary["kernel_errors"] else ["- None observed"]),
            "",
            "## Disposition",
            "",
            "- AUDIT-01 + AUDIT-03 both PASS: integrated runtime viable; Phase 4 cites these as supporting evidence.",
            "- Either fails: Phase 4 documents the specific failure mode + the conservative Strix Halo derate it implies.",
        ]
        md_path.write_text("\n".join(md))


    def main():
        import argparse
        ap = argparse.ArgumentParser()
        ap.add_argument("--duration-s", type=int, default=DEFAULT_DURATION_S)
        ap.add_argument("--concurrency", type=int, default=DEFAULT_CONCURRENCY)
        ap.add_argument("--swap-at-s", type=int, default=SWAP_AT_S)
        ap.add_argument("--run-id", default=None)
        args = ap.parse_args()
        logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
        result = asyncio.run(run_audit(
            duration_s=args.duration_s, concurrency=args.concurrency,
            swap_at_s=args.swap_at_s, run_id=args.run_id,
        ))
        raise SystemExit(0 if (result["audit_01_pass"] and result["audit_03_pass"]) else 1)


    if __name__ == "__main__":
        main()
    ```
