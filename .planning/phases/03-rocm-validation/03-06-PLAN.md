    **Part B — `tools/audit_op_coverage.py`:**

    Implements Strategy B from RESEARCH Open Q 2: profiler captured at TWO levels — model forward pass run directly in PyTorch (for aten-level op set; portable to gfx1151) AND inside vLLM serve (for the actual production op set on gfx942).

    ```python
    """AUDIT-02 — gfx1151 op coverage audit.

    Captures torch.profiler trace per model, extracts op set, cross-references against
    audit/gfx1151_registry.yaml. Writes audit/gfx1151_op_status.md.

    Strategy: capture profiler trace at TWO levels per Open Q 2:
      (a) Direct PyTorch forward pass — aten op set portable to gfx1151
      (b) vLLM serve invocation — actual production op set on gfx942 (may include
          AITER kernels that bypass aten)
    """
    from __future__ import annotations
    import argparse
    import json
    import logging
    import pathlib
    from typing import Callable

    import yaml

    logger = logging.getLogger(__name__)


    def capture_model_ops(model_name: str, warm_call_fn: Callable[[], None],
                        output_dir: pathlib.Path, double_warmup: bool = True) -> set[tuple[str, str]]:
        """Run warm_call_fn once outside profile, then once inside; export Chrome trace.

        Returns the set of distinct (category, name) pairs from the trace.
        """
        import torch
        from torch.profiler import profile, ProfilerActivity, record_function

        # Warmup: JIT / compile / cache fills happen here, not in the trace
        if double_warmup:
            warm_call_fn()

        output_dir.mkdir(parents=True, exist_ok=True)
        trace_path = output_dir / f"{model_name}_ops.json"

        with profile(
            activities=[ProfilerActivity.CPU, ProfilerActivity.CUDA],
            record_shapes=True,
            with_stack=False,
        ) as prof:
            with record_function(f"{model_name}_inference"):
                warm_call_fn()
        prof.export_chrome_trace(str(trace_path))
        logger.info(f"[audit-02] {model_name} trace at {trace_path}")
        return extract_ops_from_trace(trace_path)


    def extract_ops_from_trace(trace_path: pathlib.Path) -> set[tuple[str, str]]:
        with open(trace_path) as f:
            trace = json.load(f)
        ops: set[tuple[str, str]] = set()
        for ev in trace.get("traceEvents", []):
            name = (ev.get("name") or "").strip()
            cat = (ev.get("cat") or "").strip()
            if cat in ("kernel", "gpu_op", "cpu_op") and name:
                ops.add((cat, name))
        return ops


    def cross_reference(captured_per_model: dict[str, set[tuple[str, str]]],
                        registry: dict) -> list[dict]:
        """Build cross-ref rows. Each row = one op (or one captured but unregistered op)."""
        registry_by_name: dict[str, dict] = {entry["op"]: entry for entry in registry.get("ops", [])}

        # First: every registry op gets a row, with used_by populated from what captured it
        rows: list[dict] = []
        for op_name, entry in registry_by_name.items():
            used_by = sorted([model for model, ops in captured_per_model.items()
                            if any(name == op_name for _, name in ops)])
            rows.append({
                "op": op_name,
                "used_by": ", ".join(used_by) if used_by else "(not observed in this run)",
                "gfx942_status": entry.get("gfx942_status", "unknown"),
                "gfx1151_status": entry.get("gfx1151_status", "unknown"),
                "source": entry.get("source", ""),
                "derate_hint": entry.get("derate_hint", ""),
                "notes": entry.get("notes", "") or "",
            })

        # Second: ops captured but NOT in registry (unknown gfx1151 status)
        all_captured_names: set[str] = set()
        for ops in captured_per_model.values():
            all_captured_names.update(name for _, name in ops)
        unknown_names = all_captured_names - set(registry_by_name.keys())
        for op_name in sorted(unknown_names):
            used_by = sorted([model for model, ops in captured_per_model.items()
                            if any(name == op_name for _, name in ops)])
            rows.append({
                "op": op_name,
                "used_by": ", ".join(used_by),
                "gfx942_status": "captured-on-MI300X",
                "gfx1151_status": "unknown",
                "source": "(not in registry; needs research)",
                "derate_hint": "Unknown status; treat as RISK in Phase 4 derate model",
                "notes": "",
            })

        return rows


    def write_op_status_md(rows: list[dict], output_path: pathlib.Path) -> None:
        lines = [
            "# AUDIT-02: gfx1151 Op Coverage Status",
            "",
            "Sources: torch.profiler trace on MI300X (gfx942) cross-referenced against `audit/gfx1151_registry.yaml`.",
            "Registry sources: PyTorch #171687, ROCm/ROCm #6034, hipBLASLt #1243, ROCm/TheRock discussions/655.",
            "",
            "Status legend:",
            "- `native` — tuned kernel exists on this arch",
            "- `fallback` — runs via slower path",
            "- `unknown` — not in registry; flag as RISK in Phase 4",
            "- `KNOWN_BUG` — documented broken on this arch",
            "",
            "| Op (aten name) | Used By | gfx942 Status | gfx1151 Status | Source | Derate Hint |",
            "|----------------|---------|---------------|----------------|--------|-------------|",
        ]
        for r in sorted(rows, key=lambda x: (x["gfx1151_status"] != "native",
                                            x["gfx1151_status"] != "fallback",
                                            x["op"])):
            lines.append(
                f"| `{r['op']}` | {r['used_by']} | {r['gfx942_status']} | {r['gfx1151_status']} | {r['source']} | {r['derate_hint']} |"
            )
        lines.extend([
            "",
            "## Phase 4 Hand-off",
            "",
            "- The `Derate Hint` column feeds `derating/strix_model.py`'s per-op compute-bound vs bandwidth-bound classification (CLAUDE.md §7.3).",
            "- Every `unknown` op is a RISK to flag in the synthesis 'What We Did Not Measure' section.",
            "- The `KNOWN_BUG` rows (bf16 layer_norm) imply a conservative derate; consider fp16 fallback path on appliance.",
            "- The `fallback` rows for `aten::mm`/`aten::addmm`/`aten::bmm` imply ~3-5× extra slowdown on gemm-heavy stages beyond bandwidth ratio (Pitfall 2).",
            "- `aten::copy_` flagged per #171687 — hipMemcpyWithStream dominance on gfx1151 is the load-bearing finding NOT captured by CLAUDE.md §7.2's bandwidth-only derate.",
        ])
        output_path.write_text("\n".join(lines))


    # ---- Warm-call functions (one per model) ---------------------------

    def _warm_call_whisper():
        """Run a single distil-whisper inference via faster-whisper (CTranslate2 ROCm)."""
        from faster_whisper import WhisperModel  # type: ignore[import-untyped]
        import numpy as np
        model = WhisperModel("/models/whisper", device="cuda", compute_type="int8")
        audio = np.zeros(16000, dtype=np.float32)  # 1 sec silence
        list(model.transcribe(audio, beam_size=1, vad_filter=False)[0])

    def _warm_call_qwen():
        """Run a single Qwen3-4B forward pass via vLLM's local LLM API."""
        from vllm import LLM, SamplingParams  # type: ignore[import-untyped]
        llm = LLM(model="/models/qwen3_4b_awq", quantization="awq", dtype="auto",
                max_model_len=512, max_num_seqs=1, enforce_eager=True)
        llm.generate(["Hello world"], SamplingParams(max_tokens=8))

    def _warm_call_chatterbox():
        """Single Chatterbox render via HTTP (container assumed running)."""
        import httpx
        httpx.post("http://localhost:8004/v1/audio/speech",
                json={"input": "hello", "voice": "default"}, timeout=30.0)

    def _warm_call_kokoro():
        """Single Kokoro render via HTTP (container assumed running)."""
        import httpx
        httpx.post("http://localhost:8005/v1/audio/speech",
                json={"input": "hello", "voice": "af_alloy"}, timeout=30.0)


    WARM_CALLS = {
        "whisper": _warm_call_whisper,
        "qwen3_4b": _warm_call_qwen,
        "chatterbox": _warm_call_chatterbox,
        "kokoro": _warm_call_kokoro,
    }


    def main():
        ap = argparse.ArgumentParser()
        ap.add_argument("--models", default="whisper,qwen3_4b,chatterbox,kokoro",
                        help="comma-separated subset")
        ap.add_argument("--output-dir", default="results/audit_02")
        ap.add_argument("--registry", default="audit/gfx1151_registry.yaml")

        captured: dict[str, set[tuple[str, str]]] = {}
        for model_name in args.models.split(","):
            model_name = model_name.strip()
            if model_name not in WARM_CALLS:
                logger.warning(f"[audit-02] unknown model {model_name!r}; skipping")
                continue
            try:
                captured[model_name] = capture_model_ops(model_name, WARM_CALLS[model_name], output_dir)
            except Exception as e:
                logger.error(f"[audit-02] {model_name} capture failed: {type(e).__name__}: {e}")
                captured[model_name] = set()

        rows = cross_reference(captured, registry)
        write_op_status_md(rows, pathlib.Path(args.status_md))
        logger.info(f"[audit-02] wrote {args.status_md} with {len(rows)} op rows")


    if __name__ == "__main__":
        main()
    ```
