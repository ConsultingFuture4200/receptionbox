    ```python
    """GATE-CHATTERBOX-D1 — Day-1 ROCm load smoke for Chatterbox-Turbo (D-35).

    Runs 4 checks on a Vultr MI300X pod. Writes audit/chatterbox_d1_decision.md
    + audit/chatterbox_d1_decision.json. Drives the D-37 tts.primary flip.

    DOES NOT measure latency — that's G7. D-36 hard caps: 2-hr wall-clock, $4 spend
    (enforced by orchestration/vultr_mi300x.provision max_minutes=120 + projected_cost=4.0).
    """
    from __future__ import annotations
    import dataclasses
    import datetime
    import hashlib
    import io
    import json
    import logging
    import os
    import pathlib
    import subprocess
    import time
    from typing import Any

    import httpx
    import soundfile as sf

    logger = logging.getLogger(__name__)

    TEST_PROMPT = "The quick brown fox jumps over the lazy dog. " * 3
    CHATTERBOX_URL = os.environ.get("CHATTERBOX_URL", "http://localhost:8004")
    CHATTERBOX_CONTAINER = os.environ.get("CHATTERBOX_CONTAINER", "chatterbox")


    @dataclasses.dataclass(frozen=True)
    class CheckResult:
        ok: bool
        detail: str


    @dataclasses.dataclass(frozen=True)
    class Decision:
        pass_: bool
        reasons: list[str]
        check_1: CheckResult
        check_2: CheckResult
        check_3: CheckResult
        check_4: CheckResult
        test_prompt: str
        output_wav_sha: str | None
        timestamp_utc: str


    def check_1_container_started() -> CheckResult:
        """Verify Chatterbox container is running and reports GPU count > 0.

        Pitfall 1: rocminfo can succeed while torch.cuda.device_count()==0
        due to torch/ROCm version mismatch. Check BOTH.
        """
        try:
            ps = subprocess.run(
                ["docker", "ps", "--filter", f"name={CHATTERBOX_CONTAINER}", "--format", "{{.ID}}"],
                capture_output=True, text=True, timeout=10,
            )
            cid = (ps.stdout or "").strip().splitlines()
            if not cid:
                return CheckResult(False, f"no running container matching name={CHATTERBOX_CONTAINER}")

            # Probe both rocminfo and torch.cuda.device_count() inside the container
            rocm_probe = subprocess.run(
                ["docker", "exec", cid[0], "rocminfo"],
                capture_output=True, text=True, timeout=15,
            )
            rocm_ok = rocm_probe.returncode == 0 and "MI300" in (rocm_probe.stdout or "")

            torch_probe = subprocess.run(
                ["docker", "exec", cid[0], "python", "-c", "import torch; print(torch.cuda.device_count())"],
                capture_output=True, text=True, timeout=15,
            )
            count = int((torch_probe.stdout or "0").strip() or 0)

            if rocm_ok and count > 0:
                return CheckResult(True, f"rocm_ok=True device_count={count}")
            if rocm_ok and count == 0:
                return CheckResult(False, "device_count=0 despite rocminfo OK (Pitfall 1; devnen #92)")
            return CheckResult(False, f"rocminfo failed rc={rocm_probe.returncode}")
        except Exception as e:
            return CheckResult(False, f"check_1 exception: {type(e).__name__}: {e}")


    def check_2_endpoint_responds_in_60s() -> tuple[CheckResult, bytes]:
        start = time.monotonic()
        try:
            r = httpx.post(
                f"{CHATTERBOX_URL}/v1/audio/speech",
                json={"input": TEST_PROMPT, "voice": "default"},
                timeout=60.0,
            )
            elapsed = time.monotonic() - start
            ok = r.status_code == 200 and elapsed < 60.0
            return (CheckResult(ok, f"status={r.status_code} elapsed={elapsed:.1f}s bytes={len(r.content)}"), r.content)
        except Exception as e:
            return (CheckResult(False, f"exception: {type(e).__name__}: {e}"), b"")


    def check_3_pcm_valid(audio_bytes: bytes) -> CheckResult:
        if not audio_bytes:
            return CheckResult(False, "no audio bytes returned by check 2")
        try:
            data, sr = sf.read(io.BytesIO(audio_bytes))
            rms = float((data ** 2).mean() ** 0.5)
            dur = len(data) / sr if sr else 0.0
            ok = rms > 0.01 and dur > 1.0
            return CheckResult(ok, f"sr={sr} rms={rms:.3f} dur={dur:.2f}s")
        except Exception as e:
            return CheckResult(False, f"sf.read failed: {type(e).__name__}: {e}")


    def check_4_no_exceptions_in_logs() -> CheckResult:
        try:
            logs = subprocess.run(
                ["docker", "logs", "--since", "30s", CHATTERBOX_CONTAINER],
                capture_output=True, text=True, timeout=10,
            )
            combined = (logs.stdout or "") + (logs.stderr or "")
            hits = []
            for kw in ("Traceback (most recent call last)", "Exception:", "Error:", "FATAL"):
                if kw in combined:
                    # Capture the line and 5 lines after for diagnostic
                    idx = combined.find(kw)
                    hits.append(combined[idx : idx + 400])
            if hits:
                return CheckResult(False, f"{len(hits)} exception/traceback found: {hits[0][:200]!r}")
            return CheckResult(True, "0 exceptions in 30s window")
        except Exception as e:
            return CheckResult(False, f"check_4 exception: {type(e).__name__}: {e}")


    def decide(c1: CheckResult, c2: CheckResult, c3: CheckResult, c4: CheckResult, audio_sha: str | None) -> Decision:
        reasons = []
        for i, ck in enumerate([c1, c2, c3, c4], start=1):
            tag = "PASS" if ck.ok else "FAIL"
            reasons.append(f"check_{i}: {tag} — {ck.detail}")
        return Decision(
            pass_=all([c1.ok, c2.ok, c3.ok, c4.ok]),
            reasons=reasons,
            check_1=c1, check_2=c2, check_3=c3, check_4=c4,
            test_prompt=TEST_PROMPT,
            output_wav_sha=audio_sha,
            timestamp_utc=datetime.datetime.utcnow().isoformat(),
        )


    def write_decision(d: Decision, audit_dir: pathlib.Path, run_id: str) -> None:
        audit_dir.mkdir(parents=True, exist_ok=True)
        # 1. JSON sidecar (machine-readable; substrate may read this)
        json_path = audit_dir / "chatterbox_d1_decision.json"
        json_path.write_text(json.dumps(dataclasses.asdict(d), indent=2, default=str))
        # 2. Markdown long-form per D-38
        md_path = audit_dir / "chatterbox_d1_decision.md"
        verdict = "PASS" if d.pass_ else "FAIL"
        md = [
            "# Chatterbox-Turbo ROCm Day-1 Decision (GATE-CHATTERBOX-D1)",
            "",
            f"**Run ID:** {run_id}",
            f"**Timestamp:** {d.timestamp_utc}",
            f"**Decision:** {verdict}",
            "",
            "## D-35 Pass Criteria",
            "",
            *(f"- {r}" for r in d.reasons),
            "",
            "## Test Prompt",
            "```",
            d.test_prompt,
            "```",
            "",
            "## Output WAV SHA-256",
            f"`{d.output_wav_sha or '(no audio output)'}`",
            "",
            "## Disposition",
            "",
            f"- **tts.primary** flips to: `{'chatterbox' if d.pass_ else 'kokoro'}` per D-37.",
            "- **DR-27** fallback (Chatterbox unhealthy → Kokoro) remains wired on top of whichever is primary.",
            f"- **Downstream:** Plan 03-03 (G1/G2/G3/G5) and 03-04 (G7) will measure with primary=`{'chatterbox' if d.pass_ else 'kokoro'}`.",
            "",
            "## Install Commands Tried",
            "",
            "(populated from pod_entrypoint.sh chatterbox_d1 dispatch + container log capture)",
        ]
        md_path.write_text("\n".join(md))


    def run() -> int:
        run_id = os.environ.get("RUN_ID", datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ"))
        audit_dir = pathlib.Path(os.environ.get("AUDIT_DIR", "/workspace/audit"))
        logger.info(f"[chatterbox-d1] starting smoke run_id={run_id}")
        c1 = check_1_container_started()
        logger.info(f"[chatterbox-d1] check_1: {c1.ok} — {c1.detail}")
        c2, audio = check_2_endpoint_responds_in_60s()
        logger.info(f"[chatterbox-d1] check_2: {c2.ok} — {c2.detail}")
        c3 = check_3_pcm_valid(audio)
        logger.info(f"[chatterbox-d1] check_3: {c3.ok} — {c3.detail}")
        c4 = check_4_no_exceptions_in_logs()
        logger.info(f"[chatterbox-d1] check_4: {c4.ok} — {c4.detail}")
        audio_sha = hashlib.sha256(audio).hexdigest() if audio else None
        d = decide(c1, c2, c3, c4, audio_sha)
        write_decision(d, audit_dir, run_id)
        logger.info(f"[chatterbox-d1] DECISION: {'PASS' if d.pass_ else 'FAIL'}")
        return 0 if d.pass_ else 1


    if __name__ == "__main__":
        logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
        raise SystemExit(run())
    ```
