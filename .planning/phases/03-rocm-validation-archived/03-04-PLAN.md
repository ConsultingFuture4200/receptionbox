GATE-G7 produces measurement data for the TTS A/B preference test on MI300X. Per CLAUDE.md §5.3 and §5.4, the gate has two distinct measurements:

1. **First-audio latency** — time from `POST /v1/audio/speech` to first PCM chunk return; measured WARM (engine container already loaded; same Python process) AND COLD (container restarted between renders, simulating thermal/cache reset). Both engines (Chatterbox + Kokoro) measured under both thermal conditions.

2. **30 stimulus pairs rendered** — same 30 utterances (legal terms, proper nouns, numbers, edge cases from Phase 1 ASSETS-06) rendered through both engines for the offline blind listening test (Phase 4 deliverable).

Phase 4 derates these warm and cold numbers separately to Strix Halo, then synthesis runs the blind listening test offline.

Output: results/g7/{run_id}.jsonl with 120+ rows (30 pairs × 2 engines × 2 thermal states); audio WAVs saved under `results/g7/audio/{engine}/{pair_id}_{warm|cold}.wav` for the blind listening test.

    ```python
    """GATE-G7 — TTS A/B preference render (Chatterbox + Kokoro, warm + cold).

    30 stimulus pairs from Phase 1 ASSETS-06. Each rendered through both engines
    under WARM (serial in same process) and COLD (container restart) conditions.

    Per CLAUDE.md §5.3 + §5.4:
    - First-audio measured at first PCM chunk return from streaming endpoint.
    - Audio loudness-normalized to -23 LUFS via pyloudnorm for fair A/B.
    - 30 pairs × 2 engines × 2 thermal = 120 rows minimum.
    """
    from __future__ import annotations
    import asyncio
    import dataclasses
    import hashlib
    import io
    import json
    import logging
    import pathlib
    import subprocess
    import time
    from collections.abc import Iterable
    from typing import Any

    import httpx
    import numpy as np
    import pyloudnorm as pyln
    import soundfile as sf

    from gates._runner_base import build_result, write_jsonl

    logger = logging.getLogger(__name__)

    STIMULUS_KINDS = {"numbers", "proper_nouns", "legal_term", "edge_case", "generic"}
    G7_ENGINES = ("chatterbox", "kokoro")
    G7_THERMAL = ("warm", "cold")
    G7_AUDIO_DIR = pathlib.Path("results/g7/audio")
    G7_TARGET_LUFS = -23.0


    @dataclasses.dataclass(frozen=True)
    class StimulusPair:
        pair_id: int
        text: str
        reference_wav_path: pathlib.Path | None  # for Chatterbox voice clone
        stimulus_kind: str


    def _load_pairs(pairs_dir: pathlib.Path) -> list[StimulusPair]:
        pairs = []
        for meta_path in sorted(pairs_dir.glob("pair_*/metadata.json")):
            meta = json.loads(meta_path.read_text())
            stim_text = (meta_path.parent / "stimulus.txt").read_text().strip()
            ref_wav = meta_path.parent / "reference.wav"
            pairs.append(StimulusPair(
                pair_id=int(meta["pair_id"]),
                text=stim_text,
                reference_wav_path=ref_wav if ref_wav.exists() else None,
                stimulus_kind=meta.get("stimulus_kind", "generic"),
            ))
        return pairs


    async def _render_warm(client_url: str, stimulus: str, voice_ref: pathlib.Path | None) -> tuple[float, float, bytes]:
        """POST /v1/audio/speech streaming; measure first_audio_ms + total_ms.

        Returns (first_audio_ms, total_ms, audio_bytes).
        """
        t0 = time.monotonic()
        first_chunk_ts: float | None = None
        buf = bytearray()
        try:
            async with httpx.AsyncClient(timeout=60.0) as cli:
                async with cli.stream(
                    "POST",
                    f"{client_url}/v1/audio/speech",
                    json={"input": stimulus, "voice": "default"},
                ) as r:
                    r.raise_for_status()
                    async for chunk in r.aiter_bytes():
                        if first_chunk_ts is None and chunk:
                            first_chunk_ts = time.monotonic()
                        buf.extend(chunk)
        except Exception as e:
            logger.warning(f"[g7-warm] render failed: {type(e).__name__}: {e}")
            raise
        total_ms = (time.monotonic() - t0) * 1000.0
        first_ms = ((first_chunk_ts or time.monotonic()) - t0) * 1000.0
        return (first_ms, total_ms, bytes(buf))


    async def _render_cold(engine_name: str, container_name: str, client_url: str,
                          stimulus: str, voice_ref: pathlib.Path | None) -> tuple[float, float, bytes]:
        """Restart container, wait for /health, then measure first_audio_ms.

        Returns (first_audio_ms, total_ms, audio_bytes).
        """
        # Restart the container — measures cold cache start
        subprocess.run(["docker", "restart", container_name], check=False, timeout=30)
        # Wait for /health up to 90s
        for _ in range(45):
            try:
                r = httpx.get(f"{client_url}/health", timeout=2.0)
                if r.status_code == 200:
                    break
            except Exception:
                pass
            await asyncio.sleep(2)
        else:
            raise RuntimeError(f"[g7-cold] {engine_name} did not become healthy after restart")
        # Now render — this measures first-audio cold from a freshly-restarted container
        return await _render_warm(client_url, stimulus, voice_ref)


    def _analyze_audio(audio_bytes: bytes) -> dict[str, Any]:
        """Return (audio_rms, duration_ms, sha256) + loudness-normalized WAV bytes."""
        try:
            data, sr = sf.read(io.BytesIO(audio_bytes))
            if data.ndim > 1:
                data = data.mean(axis=1)  # mono
            duration_ms = int(1000 * len(data) / sr)
            rms = float(np.sqrt((data ** 2).mean()))
            meter = pyln.Meter(sr)
            loudness = meter.integrated_loudness(data)
            normalized = pyln.normalize.loudness(data, loudness, G7_TARGET_LUFS)
            wav_buf = io.BytesIO()
            sf.write(wav_buf, normalized, sr, format="WAV")
            sha = hashlib.sha256(audio_bytes).hexdigest()
            return {
                "audio_rms": rms, "audio_duration_ms": duration_ms,
                "audio_sha256": sha, "lufs_raw": loudness,
                "normalized_wav_bytes": wav_buf.getvalue(),
            }
        except Exception as e:
            logger.warning(f"[g7] audio analysis failed: {type(e).__name__}: {e}")
            return {"audio_rms": None, "audio_duration_ms": None, "audio_sha256": None,
                    "lufs_raw": None, "normalized_wav_bytes": None, "error": str(e)}


    async def run_g7(*, chatterbox_url: str, kokoro_url: str,
                    chatterbox_container: str, kokoro_container: str,
                    pairs_dir: pathlib.Path, run_id: str,
                    feasibility_risk_chatterbox: bool) -> int:
        pairs = _load_pairs(pairs_dir)
        if not pairs:
            logger.error(f"[g7] no stimulus pairs found at {pairs_dir}")
            return 1
        G7_AUDIO_DIR.mkdir(parents=True, exist_ok=True)
        for eng in G7_ENGINES:
            (G7_AUDIO_DIR / eng).mkdir(parents=True, exist_ok=True)

        urls = {"chatterbox": chatterbox_url, "kokoro": kokoro_url}
        containers = {"chatterbox": chatterbox_container, "kokoro": kokoro_container}

        for pair in pairs:
            for engine in G7_ENGINES:
                for thermal in G7_THERMAL:
                    try:
                        if thermal == "warm":
                            first_ms, total_ms, audio = await _render_warm(urls[engine], pair.text, pair.reference_wav_path)
                        else:
                            first_ms, total_ms, audio = await _render_cold(engine, containers[engine], urls[engine], pair.text, pair.reference_wav_path)
                        analysis = _analyze_audio(audio)
                        # Save normalized WAV for blind listening test
                        wav_path = G7_AUDIO_DIR / engine / f"{pair.pair_id:02d}_{thermal}.wav"
                        if analysis.get("normalized_wav_bytes"):
                            wav_path.write_bytes(analysis["normalized_wav_bytes"])
                        extras = {
                            "engine": engine, "warm_or_cold": thermal,
                            "pair_id": pair.pair_id, "stimulus_kind": pair.stimulus_kind,
                        }
                        if engine == "chatterbox" and feasibility_risk_chatterbox:
                            extras["feasibility_risk"] = True
                        row = build_result(
                            gate="g7", substrate="rocm",
                            asset_id=f"g7_pair_{pair.pair_id:02d}",
                            metrics={
                                "first_audio_ms": first_ms,
                                "render_total_ms": total_ms,
                                "audio_rms": analysis.get("audio_rms"),
                                "audio_duration_ms": analysis.get("audio_duration_ms"),
                                "audio_sha256": analysis.get("audio_sha256"),
                                "lufs_raw": analysis.get("lufs_raw"),
                            },
                            extras=extras, status="ok", run_id=run_id,
                        )
                    except Exception as e:
                        logger.warning(f"[g7] engine={engine} thermal={thermal} pair={pair.pair_id} failed: {e}")
                        extras = {"engine": engine, "warm_or_cold": thermal,
                                "pair_id": pair.pair_id, "stimulus_kind": pair.stimulus_kind,
                                "error": f"{type(e).__name__}: {e}"}
                        if engine == "chatterbox" and feasibility_risk_chatterbox:
                            extras["feasibility_risk"] = True
                        row = build_result(
                            gate="g7", substrate="rocm",
                            asset_id=f"g7_pair_{pair.pair_id:02d}",
                            metrics={}, extras=extras, status="error", run_id=run_id,
                        )
                    write_jsonl(row, f"results/g7/{run_id}.jsonl")
        return 0


    def _read_feasibility_risk_flag() -> bool:
        path = pathlib.Path("audit/chatterbox_d1_decision.json")
        if not path.exists():
            return False
        try:
            d = json.loads(path.read_text())
            return not d.get("pass_", True)
        except Exception:
            return False


    def main():
        import argparse, os
        ap = argparse.ArgumentParser()
        ap.add_argument("--run-id", default=os.environ.get("RUN_ID", f"g7_{int(time.time())}"))
        ap.add_argument("--pairs-dir", default="assets/g7_pairs")
        ap.add_argument("--chatterbox-url", default=os.environ.get("CHATTERBOX_URL", "http://localhost:8004"))
        ap.add_argument("--kokoro-url", default=os.environ.get("KOKORO_URL", "http://localhost:8005"))
        ap.add_argument("--chatterbox-container", default="chatterbox")
        ap.add_argument("--kokoro-container", default="kokoro")
        args = ap.parse_args()

        risk = _read_feasibility_risk_flag()
        rc = asyncio.run(run_g7(
            chatterbox_url=args.chatterbox_url, kokoro_url=args.kokoro_url,
            chatterbox_container=args.chatterbox_container,
            kokoro_container=args.kokoro_container,
            pairs_dir=pathlib.Path(args.pairs_dir), run_id=args.run_id,
            feasibility_risk_chatterbox=risk,
        ))
        raise SystemExit(rc)


    if __name__ == "__main__":
        main()
    ```
