    ```python
    """ONNX Runtime ROCm STT engine — parallel decoder for G2 dual-path (D-Discretion).

    Distil-whisper-large-v3 exported via optimum.onnxruntime with encoder-decoder
    split. Same input bytes as FasterWhisperEngine; different decoder.
    Production-runtime parity per CLAUDE.md §4.2.
    """
    from __future__ import annotations
    import logging
    import pathlib
    from collections.abc import AsyncIterator

    from substrate.types import STTChunk

    logger = logging.getLogger(__name__)


    class ONNXWhisperEngine:
        def __init__(self, *, model_dir: str | pathlib.Path, providers: list[str] | None = None):
            self._model_dir = pathlib.Path(model_dir)
            # ROCMExecutionProvider must be first; CPUExecutionProvider as fallback
            self._providers = providers or ["ROCMExecutionProvider", "CPUExecutionProvider"]
            self._session = None
            self._processor = None
            self._loaded = False

        async def load(self) -> None:
            try:
                # Lazy imports — keep workstation-side import-time clean
                from optimum.onnxruntime import ORTModelForSpeechSeq2Seq  # type: ignore[import-untyped]
                from transformers import WhisperProcessor  # type: ignore[import-untyped]

                self._session = ORTModelForSpeechSeq2Seq.from_pretrained(
                    str(self._model_dir),
                    provider=self._providers[0],
                )
                self._processor = WhisperProcessor.from_pretrained(str(self._model_dir))
                self._loaded = True
                logger.info(f"[onnx-whisper] loaded model_dir={self._model_dir} provider={self._providers[0]}")
            except Exception as e:
                logger.warning(f"[onnx-whisper] load failed (DR-27 onnx path not available): {type(e).__name__}: {e}")
                self._loaded = False

        async def health(self) -> bool:
            return self._loaded

        async def transcribe(
            self, audio: AsyncIterator[bytes], *, sample_rate: int,
        ) -> AsyncIterator[STTChunk]:
            if not self._loaded:
                logger.warning("[onnx-whisper] transcribe called before load; yielding nothing")
                return
            # Collect audio bytes (non-streaming for INT16 PCM input)
            buf = bytearray()
            async for chunk in audio:
                buf.extend(chunk)
            import numpy as np  # noqa: E402

            # Assume 16-bit PCM little-endian (matches faster-whisper input)
            samples = np.frombuffer(bytes(buf), dtype=np.int16).astype(np.float32) / 32768.0
            inputs = self._processor(samples, sampling_rate=sample_rate, return_tensors="pt")
            generated = self._session.generate(inputs.input_features, max_new_tokens=448)
            text = self._processor.batch_decode(generated, skip_special_tokens=True)[0]
            duration_ms = int(1000 * len(samples) / sample_rate)
            yield STTChunk(text=text.strip(), start_ms=0, end_ms=duration_ms, confidence=None)
    ```

    ```python
    async def _single_e2e_call(substrate, asset, run_id, concurrency):
        # One end-to-end call: STT -> LLM -> TTS using the substrate
        # Capture timestamps via LiveKit AgentSession.metrics OR via manual time.monotonic() around each stage
        ts = {}
        ts["t0"] = time.monotonic()
        # ... STT
        stt_chunks = []
        async for c in substrate.transcribe(_load_audio(asset), sample_rate=16000):
            if not stt_chunks: ts["stt_first"] = time.monotonic()
            stt_chunks.append(c)
        ts["stt_done"] = time.monotonic()
        hyp = " ".join(c.text for c in stt_chunks)
        # ... LLM
        prompt = _build_prompt(asset.persona, asset.intent, hyp)
        llm_chunks = []
        async for c in substrate.generate(prompt, max_tokens=64):
            if not llm_chunks: ts["llm_first"] = time.monotonic()
            llm_chunks.append(c)
        ts["llm_done"] = time.monotonic()
        llm_text = "".join(c.text for c in llm_chunks)
        # ... TTS
        tts_bytes = bytearray()
        async for b in substrate.synthesize(llm_text):
            if not tts_bytes: ts["tts_first"] = time.monotonic()
            tts_bytes.extend(b)
        ts["tts_done"] = time.monotonic()
        return build_result(
            gate="g1", substrate="rocm", asset_id=asset.id,
            metrics={
                "stt_ttft_ms": int(1000 * (ts.get("stt_first", ts["stt_done"]) - ts["t0"])),
    ```

    ```python
    def _classify_constraint_status(response_obj, http_status, exception):
        if exception is not None:
            if "timeout" in str(exception).lower(): return "engine_timeout"
            return "engine_error_5xx"
        if http_status >= 500: return "engine_error_5xx"
        if http_status >= 400: return "engine_error_5xx"
        try:
            parsed = json.loads(response_obj) if isinstance(response_obj, str) else response_obj
            if not isinstance(parsed, dict) or "refusal_label" not in parsed:
                return "constraint_miss_invalid_json"
            return "constraint_ok"
        except (json.JSONDecodeError, TypeError):
            return "constraint_miss_invalid_json"

    async def run_g5(substrate, probe_set, run_id, reference_prompt_path):
        ref_prompt = pathlib.Path(reference_prompt_path).read_text()
        for probe in probe_set:
            try:
                # generate with grammar=Grammar.json_schema(...) (substrate-side xgrammar)
                chunks = []
                async for c in substrate.generate(
                    f"{ref_prompt}\n\nUser: {probe.prompt}",
                    grammar=Grammar.json_schema({"refusal_label": "string"}),  # simplified for example
                    max_tokens=128,
                ):
                    chunks.append(c.text)
                response = "".join(chunks)
                status = _classify_constraint_status(response, 200, None)
                refused = None

                    refused = parsed["refusal_label"] != "answer"
                row = build_result(
                    gate="g5", substrate="rocm", asset_id=probe.id,

                    run_id=run_id,
    ```
