       ```python
       """Build config/sanity_strata.yaml deterministically from assets/manifest.csv (D-27)."""
       from __future__ import annotations
       import argparse, csv, datetime, logging, pathlib, random, sys
       from collections import defaultdict
       import yaml

       logger = logging.getLogger(__name__)


       def _load_manifest(path: pathlib.Path) -> list[dict]:
           with path.open() as f:
               return list(csv.DictReader(f))


       def _pick_g1(rows: list[dict], rng: random.Random) -> list[str]:
           pool = [r for r in rows if r["corpus"] == "corpus_500"]
           by_intent: dict[str, list[dict]] = defaultdict(list)
           for r in pool:
               by_intent[r.get("intent", "unknown")].append(r)
           top5 = sorted(by_intent.items(), key=lambda kv: -len(kv[1]))[:5]
           picks: list[dict] = []
           for intent, group in top5:
               neutral = [r for r in group if r.get("adversity_level") == "neutral"]
               stressed = [r for r in group if r.get("adversity_level") != "neutral"]
               rng.shuffle(neutral); rng.shuffle(stressed)
               if neutral and stressed:
                   picks.extend([neutral[0], stressed[0]])
               else:
                   src = neutral or stressed
                   picks.extend(src[:2])
           # Pad if any intent had <2; pull from largest remaining.
           while len(picks) < 10 and pool:
               extra = rng.choice(pool)
               if extra not in picks:
                   picks.append(extra)
           return sorted({p["asset_id"] for p in picks[:10]})


       def _pick_g2(rows: list[dict], rng: random.Random) -> list[str]:
           pool = [r for r in rows if r["corpus"] == "corpus_g711"]
           neutral = [r for r in pool if r.get("adversity_level") == "neutral"]
           stressed = [r for r in pool if r.get("adversity_level") != "neutral"]
           rng.shuffle(neutral); rng.shuffle(stressed)
           if len(neutral) < 5:
               logger.warning(f"g2: only {len(neutral)} neutral clips available")
           if len(stressed) < 5:
               logger.warning(f"g2: only {len(stressed)} stressed clips available")
           picks = neutral[:5] + stressed[:5]
           return sorted({p["asset_id"] for p in picks})


       def _pick_g3(rows: list[dict], rng: random.Random) -> list[str]:
           pool = [r for r in rows if r["corpus"] == "corpus_hesitation"]
           by_pat: dict[str, list[dict]] = defaultdict(list)
           for r in pool:
               by_pat[r.get("hesitation_pattern", "unknown")].append(r)
           # Ensure all 4 patterns covered if present.
           patterns = ("filler_words", "mid_sentence_pause", "false_start", "mid_word_stop")
           picks: list[dict] = []
           for pat in patterns:
               group = by_pat.get(pat, [])
               rng.shuffle(group)
               picks.extend(group[:3])
           # Trim or pad to exactly 10.
           rng.shuffle(picks)
           if len(picks) > 10:
               picks = picks[:10]
           while len(picks) < 10 and pool:
               extra = rng.choice(pool)
               if extra not in picks:
                   picks.append(extra)
           return sorted({p["asset_id"] for p in picks[:10]})


       def _pick_g5(rows: list[dict], rng: random.Random) -> list[str]:
           pool = [r for r in rows if r["corpus"] == "upl_probes"]
           controls = [r for r in pool if str(r.get("control", "")).lower() in ("1","true","yes")]
           probes  = [r for r in pool if r not in controls]
           by_cat: dict[str, list[dict]] = defaultdict(list)
           for r in probes:
               by_cat[r.get("category", "unknown")].append(r)
           picks: list[dict] = []
           for cat in ("fee_quote", "statute_of_limitations", "case_outcome", "procedural_deadline"):
               group = by_cat.get(cat, [])
               rng.shuffle(group)
               picks.extend(group[:2])
           rng.shuffle(controls)
           picks.extend(controls[:2])
           return sorted({p["asset_id"] for p in picks[:10]})


       def build(*, seed: int, manifest: pathlib.Path, out: pathlib.Path) -> dict:
           rows = _load_manifest(manifest)
           # 4 independent RNGs so adding a gate doesn't perturb others.
           rng_g1 = random.Random((seed, "g1"))
           rng_g2 = random.Random((seed, "g2"))
           rng_g3 = random.Random((seed, "g3"))
           rng_g5 = random.Random((seed, "g5"))
           data = {
               "version": "1.0",
               "seed": seed,
               "generated_utc": datetime.datetime.utcnow().isoformat(),
               "manifest_path": str(manifest),
               "strata": {
                   "g1": {"rule": "10 calls; top-5 intents × 2 (1 neutral + 1 stressed when available)",
                          "assets": _pick_g1(rows, rng_g1)},
                   "g2": {"rule": "5 neutral + 5 stressed from corpus_g711",
                          "assets": _pick_g2(rows, rng_g2)},
                   "g3": {"rule": "10 hesitation clips covering 4 patterns (filler_words, mid_sentence_pause, false_start, mid_word_stop)",
                          "assets": _pick_g3(rows, rng_g3)},
                   "g5": {"rule": "8 probes (2 each × 4 refusal categories) + 2 benign controls",
                          "assets": _pick_g5(rows, rng_g5)},
               },
           }
           out.parent.mkdir(parents=True, exist_ok=True)
           out.write_text(yaml.safe_dump(data, sort_keys=False, default_flow_style=False))
           return data


       def main(argv=None) -> int:
           p = argparse.ArgumentParser()
           p.add_argument("--seed", type=int, default=42)
           p.add_argument("--manifest", type=pathlib.Path, default=pathlib.Path("assets/manifest.csv"))
           p.add_argument("--out", type=pathlib.Path, default=pathlib.Path("config/sanity_strata.yaml"))
           args = p.parse_args(argv)
           logging.basicConfig(level=logging.INFO, format="%(levelname)s %(message)s")
           data = build(seed=args.seed, manifest=args.manifest, out=args.out)
           for gate, info in data["strata"].items():
               print(f"{gate}: {len(info['assets'])} assets")
           return 0


       if __name__ == "__main__":
           sys.exit(main())
       ```

       ```python
       """Phase 2 pre-flight driver: provision → wait → validate → teardown.

       Modes:
         bootstrap — one-time HF cache pull on a small bootstrap pod
         smoke     — 5-call G1 smoke (PREFLIGHT-01)
         sanity    — sequential G1+G2+G3+G5 sanity (PREFLIGHT-02)

       Honors RUNPOD_API_KEY=unset by routing through dry-run.
       """
       from __future__ import annotations
       import argparse, asyncio, datetime, json, logging, os, pathlib, sys, time
       import yaml
       from cost.adapters import runpod as runpod_adapter
       from orchestration.runpod_h100 import provision, terminate, ProvisionResult, RunPodProvisionError
       from harness.env_sidecar import read_env_sidecar

       logger = logging.getLogger(__name__)
       H100_USD_PER_HR = 2.69  # SXM on-demand RunPod Secure (CLAUDE.md §1.1)

       def _project_cost(max_minutes: int) -> float:
           return round(max_minutes / 60.0 * H100_USD_PER_HR, 2)


       def _budget(path=pathlib.Path("config/budget.yaml")) -> dict:
           return yaml.safe_load(path.read_text())


       async def _wait_for_pod_exit(pod_id: str, timeout_s: int) -> str:
           """Poll runpod.get_pods until pod_id is gone or status==EXITED. Returns final state."""
           import httpx
           deadline = time.time() + timeout_s
           async with httpx.AsyncClient() as client:
               while time.time() < deadline:
                   spend, _projected = await runpod_adapter.poll(client)
                   logger.info(f"[preflight] cumulative_spend=${spend:.2f}")
                   try:
                       import runpod  # type: ignore[import-untyped]
                       runpod.api_key = os.environ.get("RUNPOD_API_KEY", "")
                       pods = runpod.get_pods() or []
                       found = next((p for p in pods if p.get("id") == pod_id), None)
                       if found is None:
                           return "GONE"
                       status = found.get("desiredStatus") or found.get("runtimeStatus") or "UNKNOWN"
                       if status in ("EXITED", "TERMINATED", "STOPPED"):
                           return status
                   except Exception as e:
                       logger.warning(f"[preflight] poll: {e}")
                   await asyncio.sleep(60)
           return "TIMEOUT"


       def _validate_smoke(*, results_dir: pathlib.Path, run_id_glob: str,
                            wall_clock_s: float, final_spend: float) -> dict:
           """Return D-25 verdict dict {criterion → bool}."""
           candidates = sorted(results_dir.glob(f"smoke/{run_id_glob}.jsonl"))
           if not candidates:
               return {"a_5_rows": False, "b_under_30min": False, "c_under_1usd": False,
                       "d_per_stage_timings": False, "e_env_sidecar": False, "f_audit_clean": False,
                       "error": "no JSONL found"}
           jsonl = candidates[-1]
           rows = [json.loads(line) for line in jsonl.open() if line.strip()]
           env_path = jsonl.with_suffix(".env.json")
           audit_glob = list(results_dir.glob("smoke/*.audit.json"))
           v = {
               "a_5_rows": len(rows) == 5,
               "b_under_30min": wall_clock_s < 30 * 60,
               "c_under_1usd": final_spend < 1.0,
               "d_per_stage_timings": all(
                   r.get("stt_ttft_ms") is not None and r.get("llm_ttft_ms") is not None
                   and r.get("tts_first_audio_ms") is not None and r.get("e2e_ms") is not None
                   for r in rows),
               "e_env_sidecar": env_path.exists() and bool(read_env_sidecar(env_path)),
               "f_audit_clean": (
                   audit_glob and json.loads(audit_glob[-1].read_text())["summary"]["violations"] == 0
               ),
           }
           v["pass"] = all(v[k] for k in v if k != "error")
           return v


       async def _run_gate(*, gate: str, max_minutes: int, network_volume_id: str | None,
                            ssh_pubkey: str | None, operator_host: str | None,
                            results_dir: pathlib.Path) -> dict:
           projected = _project_cost(max_minutes)
           started = time.time()
           try:
               result: ProvisionResult = provision(
                   gate=gate, projected_cost=projected,
                   max_minutes=max_minutes, network_volume_id=network_volume_id,
                   ssh_pubkey=ssh_pubkey, operator_host=operator_host,
               )
           except RunPodProvisionError as e:
               return {"gate": gate, "status": "provision_error", "error": str(e)}
           if result.pod_id == "dry-run":
               logger.info(f"[preflight] DRY RUN for gate={gate}; returning early")
               return {"gate": gate, "status": "dry-run", "pod_id": "dry-run"}
           final_state = await _wait_for_pod_exit(result.pod_id, timeout_s=max_minutes * 60 + 300)
           wall_clock_s = time.time() - started
           # Final spend reading.
           import httpx
           async with httpx.AsyncClient() as client:
               final_spend, _ = await runpod_adapter.poll(client)
           verdict = {"gate": gate, "status": final_state, "wall_clock_s": wall_clock_s,
                      "final_spend_usd": final_spend, "pod_id": result.pod_id,
                      "auth_id": result.authorization.id}
           if gate == "smoke":
               verdict["smoke_verdict"] = _validate_smoke(
                   results_dir=results_dir, run_id_glob="*",
                   wall_clock_s=wall_clock_s, final_spend=final_spend,
               )
           return verdict


       async def run(mode: str) -> int:
           cfg = _budget()
           per = cfg["phase2"]["max_minutes_per_gate"]
           results_dir = pathlib.Path("results")
           network_volume_id = os.environ.get("RUNPOD_NETWORK_VOLUME_ID")
           ssh_pubkey = os.environ.get("SSH_PUBKEY")
           operator_host = os.environ.get("OPERATOR_HOST")
           session_id = datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")
           session = {"session_id": session_id, "mode": mode, "gates": []}

           if mode == "bootstrap":
               # Provision a small pod that runs cache_bootstrap.py and exits.
               # In dry-run, just log.
               if not os.environ.get("RUNPOD_API_KEY"):
                   logger.info("[preflight] DRY RUN bootstrap")
                   return 0
               # The bootstrap pod uses a different entrypoint (just runs cache_bootstrap.py).
               # Operator sets BOOTSTRAP_MODE=1 in env; pod_entrypoint.sh checks it.
               # Implementation detail: simplest approach is a one-shot via runpodctl.
               logger.info("[preflight] bootstrap mode requires operator-side runpodctl invocation; "
                           "see docs/OPERATOR-CHECKLIST-PHASE-02.md")
               return 0

           gates = ["smoke"] if mode == "smoke" else ["g1", "g2", "g3", "g5"]
           for g in gates:
               max_min = per.get(g, 30) if g != "smoke" else per.get("smoke", 30)
               v = await _run_gate(
                   gate=g, max_minutes=max_min,
                   network_volume_id=network_volume_id,
                   ssh_pubkey=ssh_pubkey, operator_host=operator_host,
                   results_dir=results_dir,
               )
               session["gates"].append(v)
               if mode == "smoke" and v.get("smoke_verdict", {}).get("pass") is False:
                   logger.error(f"[preflight] smoke FAILED: {v['smoke_verdict']}")
                   break

           out = results_dir / "preflight" / f"{session_id}.json"
           out.parent.mkdir(parents=True, exist_ok=True)
           out.write_text(json.dumps(session, indent=2, sort_keys=True, default=str))
           print(json.dumps(session, indent=2, default=str))
           # Exit 1 if any gate failed.
           any_fail = any(g.get("status") not in ("EXITED", "GONE", "TERMINATED", "dry-run", "STOPPED")
                          for g in session["gates"])
           return 1 if any_fail else 0


       def main(argv=None) -> int:
           p = argparse.ArgumentParser()
           p.add_argument("--mode", choices=["bootstrap", "smoke", "sanity"], required=True)
           args = p.parse_args(argv)
           logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
           return asyncio.run(run(args.mode))


       if __name__ == "__main__":
           sys.exit(main())
       ```
