    ```python
    """Orin-target derate model (DERATE-02, DERATE-05).

    Companion to derating/strix_model.py (Phase 0 archive — Strix Halo target,
    pre-DR-39 pivot). Numbers and ratios from CLAUDE.md §7 (DR-39 v0.3.0).
    """

    from __future__ import annotations
    from dataclasses import dataclass
    from collections.abc import Iterable
    from .op_classes import HardwareSpec, OpClass, StageMeasurement

    # CLAUDE.md §7.1 spec sheets
    @dataclass(frozen=True)
    class OrinSpec(HardwareSpec):  # may need to be a separate dataclass if HardwareSpec is closed
        fp16_tflops_sparse: float
        int8_tops_sparse: float
        power_w_max: int

    H100_PCIE = OrinSpec(
        name="H100 PCIe 80GB",
        bandwidth_gb_s=2000.0,
        prompt_processing_factor=1.0,
        fp16_tflops_sparse=756.0,
        int8_tops_sparse=3026.0,
        power_w_max=350,
    )

    ORIN_AGX_64GB = OrinSpec(
        name="Jetson AGX Orin 64GB",
        bandwidth_gb_s=204.0,
        prompt_processing_factor=23.6,  # FP16 ratio H100_PCIE / Orin = 756/32 ≈ 23.6
        fp16_tflops_sparse=32.0,
        int8_tops_sparse=275.0,
        power_w_max=60,
    )

    DEFAULT_OLLAMA_OVERHEAD = 1.4   # CLAUDE.md §3.1 midpoint of 1.3-1.5
    DEFAULT_ARM_PENALTY = 1.15      # CLAUDE.md §7.2 midpoint of 10-20%

    def derate_compute_bound_fp16(measured_ms: float, src: OrinSpec, dst: OrinSpec) -> float:
        return measured_ms * (src.fp16_tflops_sparse / dst.fp16_tflops_sparse)

    def derate_compute_bound_int8(measured_ms: float, src: OrinSpec, dst: OrinSpec) -> float:
        return measured_ms * (src.int8_tops_sparse / dst.int8_tops_sparse)

    def derate_bandwidth_bound(measured_ms: float, src: OrinSpec, dst: OrinSpec) -> float:
        return measured_ms * (src.bandwidth_gb_s / dst.bandwidth_gb_s)

    # Stage → derate function dispatch. Phase 4 may refine per-stage classification.
    STAGE_DERATE_FUNCS = {
        "stt_ttft": derate_compute_bound_int8,     # faster-whisper INT8 encoder
        "llm_ttft": derate_compute_bound_int8,     # vLLM AWQ-Int4 prefill
        "llm_decode": derate_bandwidth_bound,      # batch=1 decode is bandwidth-bound
        "tts_first_audio": derate_compute_bound_fp16,  # TTS prefill is FP16
    }

    def derate_pipeline(
        measurements: Iterable[StageMeasurement],
        src: OrinSpec = H100_PCIE,
        dst: OrinSpec = ORIN_AGX_64GB,
        ollama_overhead: float | None = None,
        arm_penalty: float | None = None,
    ) -> dict:
        oo = DEFAULT_OLLAMA_OVERHEAD if ollama_overhead is None else ollama_overhead
        ap = DEFAULT_ARM_PENALTY if arm_penalty is None else arm_penalty
        out: dict = {"_ollama_overhead": oo, "_arm_penalty": ap}
        for m in measurements:
            fn = STAGE_DERATE_FUNCS.get(m.stage)
            if fn is None:
                out[m.stage] = None
                continue
            base = fn(m.measured_ms, src, dst)
            if m.stage in ("llm_ttft", "llm_decode"):
                base *= oo  # Ollama overhead on LLM stages
            base *= ap  # ARM penalty universal
            out[m.stage] = base
        out["total_ms"] = sum(v for k, v in out.items()
                              if k not in ("_ollama_overhead","_arm_penalty","total_ms")
                              and v is not None)
        return out
    ```

    ```python
    # synthesis/derate_pipeline.py
    import json, pathlib
    import pandas as pd
    import numpy as np
    from scipy import stats
    from derating.orin_model import (
        H100_PCIE, ORIN_AGX_64GB,
        STAGE_DERATE_FUNCS, DEFAULT_OLLAMA_OVERHEAD, DEFAULT_ARM_PENALTY,
    )

    def _measure_ollama_overhead() -> float | None:
        path = next(pathlib.Path("results/audit_03").glob("*.jsonl"), None)
        if not path: return None
        rows = [json.loads(l) for l in path.read_text().splitlines() if l.strip()]
        vllm_tps = [r["metrics"]["tokens_per_sec"] for r in rows
                    if r["metrics"].get("engine_kind") == "vllm"
                    and r["metrics"].get("tokens_per_sec")]
        olm_tps = [r["metrics"]["tokens_per_sec"] for r in rows
                   if r["metrics"].get("engine_kind") == "ollama"
                   and r["metrics"].get("tokens_per_sec")]
        if not vllm_tps or not olm_tps: return None
        return float(np.median(vllm_tps) / np.median(olm_tps))

    def _bootstrap_ci(samples: np.ndarray, fn=np.median, n_resamples: int = 10000) -> tuple[float, float, float]:
        if len(samples) < 2: return (float(samples.mean()), float(samples.min()), float(samples.max()))
        r = stats.bootstrap((samples,), fn, n_resamples=n_resamples,
                            confidence_level=0.95, method="percentile")
        return (float(fn(samples)), float(r.confidence_interval.low),
                float(r.confidence_interval.high))

    STAGE_COLS = {
        "stt_ttft": "stt_ttft_ms", "llm_ttft": "llm_ttft_ms",
        "llm_decode": "llm_decode_ms_per_tok",
        "tts_first_audio": "tts_first_audio_ms",
    }

    def run(df: pd.DataFrame, src=H100_PCIE, dst=ORIN_AGX_64GB) -> pd.DataFrame:
        oo = _measure_ollama_overhead() or DEFAULT_OLLAMA_OVERHEAD
        ap = DEFAULT_ARM_PENALTY
        out_rows = []
        for stage, col in STAGE_COLS.items():
            if col not in df.columns: continue
            for (gate, concurrency), grp in df.groupby(["gate","concurrency"], dropna=False):
                samples = grp[col].dropna().values
                if len(samples) == 0: continue
                h100_p50, _, _ = _bootstrap_ci(samples)
                derate_fn = STAGE_DERATE_FUNCS.get(stage)
                if derate_fn is None: continue
                # Derate every sample, then bootstrap CI on derated samples
                derated = np.array([derate_fn(s, src, dst) for s in samples])
                if stage.startswith("llm"): derated *= oo
                derated *= ap
                orin_p50, orin_lo, orin_hi = _bootstrap_ci(derated)
                out_rows.append({
                    "gate": gate, "stage": stage, "concurrency": concurrency,
                    "n_samples": int(len(samples)),
                    "measured_h100_p50_ms": h100_p50,
                    "derated_orin_point_ms": orin_p50,
                    "derated_orin_ci_lo_ms": orin_lo,
                    "derated_orin_ci_hi_ms": orin_hi,
                    "ollama_overhead_applied": oo if stage.startswith("llm") else 1.0,
                    "arm_penalty_applied": ap,
                })
        return pd.DataFrame(out_rows)

    def main() -> int:
        df = pd.read_parquet("results/synthesis/measurements.parquet")
        out = run(df)
        out_path = pathlib.Path("results/synthesis/orin_derate_table.csv")
        out_path.parent.mkdir(parents=True, exist_ok=True)
        out.to_csv(out_path, index=False)
        print(f"[derate] {len(out)} rows -> {out_path}")
        return 0
    ```

    ```python
    """Cross-check derated Orin predictions against NVIDIA-published benchmarks.

    Reads results/synthesis/orin_derate_table.csv + data/nvidia_orin_published_benchmarks.json.
    For each derated prediction, finds the closest analog in the NVIDIA dataset
    and computes the |divergence| = |predicted - published| / published. Flags
    any > 50% divergence.

    If the NVIDIA dataset is the unpopulated TEMPLATE (no numeric values), emits a
    nvidia_crosscheck.json with status="awaiting_operator_curation" and no flags.
    """

    import json, pathlib
    import pandas as pd

    DIVERGENCE_THRESHOLD = 0.5

    def main() -> int:
        derate_path = pathlib.Path("results/synthesis/orin_derate_table.csv")
        nvidia_path = pathlib.Path("data/nvidia_orin_published_benchmarks.json")
        df = pd.read_csv(derate_path)
        nvidia = json.loads(nvidia_path.read_text())
        out: dict = {"comparisons": [], "flags": [], "status": "ok"}
        published_has_data = any(
            any(v is not None for k, v in b.items() if k.startswith("value"))
            for b in nvidia.get("benchmarks", [])
        )
        if not published_has_data:
            out["status"] = "awaiting_operator_curation"
            out["note"] = ("data/nvidia_orin_published_benchmarks.json is a TEMPLATE. "
                           "Operator must populate with NVIDIA-published numbers from "
                           "developer.nvidia.com/embedded/jetson-orin-benchmarks before "
                           "the cross-check is meaningful.")
            out_path = pathlib.Path("results/synthesis/nvidia_crosscheck.json")
            out_path.write_text(json.dumps(out, indent=2))
            print(f"[crosscheck] template detected; awaiting operator data")
            return 0
        # Otherwise iterate derate rows and find matching benchmark by workload+stage
        for row in df.to_dict("records"):
            match = None
            # Workload-mapping heuristic: stage="stt_ttft" -> workload starts with "whisper"
            # stage="llm_ttft" -> workload starts with "qwen"
            for b in nvidia["benchmarks"]:
                if (row["stage"] == "stt_ttft" and b["workload"].startswith("whisper")
                    or row["stage"] in ("llm_ttft","llm_decode") and b["workload"].startswith("qwen")):
                    match = b; break
            if not match: continue
            published_val = match.get("value_ms") or match.get("value_ms_per_sec_audio") or match.get("value_tokens_per_sec")
            if published_val is None: continue
            divergence = abs(row["derated_orin_point_ms"] - published_val) / published_val
            entry = {"gate": row["gate"], "stage": row["stage"],
                     "concurrency": row["concurrency"],
                     "derated_point_ms": row["derated_orin_point_ms"],
                     "published_workload": match["workload"],
                     "published_value": published_val,
                     "divergence": divergence}
            out["comparisons"].append(entry)
            if divergence > DIVERGENCE_THRESHOLD:
                out["flags"].append({**entry, "reason": f"divergence {divergence:.1%} > {DIVERGENCE_THRESHOLD:.0%}"})
        if out["flags"]: out["status"] = "flags_present"
        out_path = pathlib.Path("results/synthesis/nvidia_crosscheck.json")
        out_path.write_text(json.dumps(out, indent=2))
        print(f"[crosscheck] status={out['status']} flags={len(out['flags'])}")
        return 0
    ```
