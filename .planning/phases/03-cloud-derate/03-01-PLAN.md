    Provisions a fresh v18 H100 pod, waits for inference services to come up,
    pings .health() on all 4 adapters via CUDASubstrate.load_*, fetches results
    back, and emits a session manifest. NO gate measurements — substrate audit only.
    """

    from harness.env_sidecar import read_env_sidecar  # may be unused; keep optional
    from orchestration.runpod_h100 import provision, terminate, ProvisionResult
    from cost.ledger import BudgetExhausted

    H100_USD_PER_HR = 2.69
    DEFAULT_MAX_MINUTES = 20  # ~$0.90 envelope; well under $5 plan cap

    def _project_cost(max_minutes: int) -> float:
        return round(max_minutes / 60.0 * H100_USD_PER_HR, 2)

    async def _wait_services_ready(pod_url: str, timeout_s: int = 600) -> tuple[bool, dict]:
        """Poll {pod_url}/vllm/v1/models + Kokoro :8005/health every 30s until ready or timeout.
        Returns (all_ready, {"vllm": bool, "kokoro": bool, ...})."""
        import httpx

        # Return after all True OR timeout.

    async def _check_substrate_health(pod_url: str) -> dict[str, bool]:
        """Construct CUDASubstrate against pod URLs and call load_stt/load_llm/load_tts.
        Read self._loaded after each load."""
        from substrate.cuda import CUDASubstrate
        sub = CUDASubstrate(
            vllm_url=f"{pod_url.replace(':8000', ':8000')}",
            vllm_model="Qwen/Qwen3-4B",
            whisper_model_dir="/models/distil_whisper_large_v3_int8",
            chatterbox_url=f"{pod_url.replace(':8000', ':8004')}",
            kokoro_url=f"{pod_url.replace(':8000', ':8005')}",
        )
        await sub.load_stt(); await sub.load_llm(); await sub.load_tts()
        return {"stt": sub._loaded["stt"], "llm": sub._loaded["llm"], "tts": sub._loaded["tts"]}

    async def main_async(argv: list[str]) -> int:
        p = argparse.ArgumentParser(prog="tools.audit_harness_health")
        p.add_argument("--real-spend", action="store_true",
                       help="Required to provision a real pod; default is dry-run.")
        p.add_argument("--max-minutes", type=int, default=DEFAULT_MAX_MINUTES)
        p.add_argument("--out-dir", type=pathlib.Path, default=pathlib.Path("results/preflight"))
        args = p.parse_args(argv)

        ts = datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")
        out_path = args.out_dir / f"03-01-harness-audit-{ts}.json"
        args.out_dir.mkdir(parents=True, exist_ok=True)

        cost = _project_cost(args.max_minutes)
        manifest: dict = {
            "plan": "03-01", "started_utc": ts, "max_minutes": args.max_minutes,
            "projected_cost_usd": cost, "real_spend": args.real_spend,
        }

        # Dry-run unless --real-spend
        if not args.real_spend or not os.environ.get("RUNPOD_API_KEY"):
            try:
                pr = provision(gate="harness-audit", projected_cost=cost,
                               max_minutes=args.max_minutes)
            except BudgetExhausted as e:
                manifest["error"] = f"BudgetExhausted: {e}"; out_path.write_text(json.dumps(manifest))
                return 2
            manifest.update({"pod_id": pr.pod_id, "image_ref": pr.image_ref,
                             "verdict": "dry-run"})
            out_path.write_text(json.dumps(manifest, indent=2, default=str))
            return 0

        # Real spend path
        try:
            pr = provision(gate="harness-audit", projected_cost=cost, max_minutes=args.max_minutes)
        except BudgetExhausted as e:
            manifest["error"] = f"BudgetExhausted: {e}"; out_path.write_text(json.dumps(manifest))
            return 2
        manifest["pod_id"] = pr.pod_id; manifest["pod_url"] = pr.pod_url
        manifest["image_ref"] = pr.image_ref
        try:
            ready, port_health = await _wait_services_ready(pr.pod_url or "")
            manifest["port_health"] = port_health
            adapter_health = await _check_substrate_health(pr.pod_url or "") if ready else {}
            manifest["adapter_health"] = adapter_health
            manifest["verdict"] = "pass" if all(adapter_health.values()) else "fail"
        finally:
            terminate(pr.pod_id)
            manifest["terminated_utc"] = datetime.datetime.utcnow().isoformat()
        out_path.write_text(json.dumps(manifest, indent=2, default=str))
        return 0 if manifest.get("verdict") == "pass" else 1

    def main() -> int:
        logging.basicConfig(level=logging.INFO)
        return asyncio.run(main_async(sys.argv[1:]))

    if __name__ == "__main__":
        sys.exit(main())
