    ```python
    """AUDIT-01 co-residency stack-load runner.

    Load all 4 models, then hold the substrate alive for `duration_s` while
    sampling VRAM every `sample_interval_s` via nvidia-smi. One result row
    per VRAM sample.

    Sustained load = the runner pokes each model with a small inference call
    once per 30s (so the GPU doesn't idle into a power-saving mode that
    would mask realistic VRAM usage).
    """

    import argparse, asyncio, logging, pathlib, subprocess, sys, time
    from gates._runner_base import GateRunner
    from harness.results import GateResult
    from substrate import Substrate

    class AUDIT01Runner(GateRunner):
        def __init__(self, *, substrate: Substrate, **kw):
            super().__init__(substrate=substrate, gate="audit_01", **kw)

        def _sample_nvidia_smi(self) -> dict:
            try:
                r = subprocess.run(
                    ["nvidia-smi",
                     "--query-gpu=memory.used,memory.total,utilization.gpu",
                     "--format=csv,noheader,nounits"],
                    capture_output=True, text=True, check=True, timeout=5,
                )
                used, total, util = [x.strip() for x in r.stdout.strip().split(",")]
                return {"vram_mb": int(used), "vram_total_mb": int(total),
                        "gpu_util_pct": int(util), "error": None}
            except (FileNotFoundError, subprocess.CalledProcessError,
                    subprocess.TimeoutExpired) as e:
                return {"vram_mb": None, "vram_total_mb": None,
                        "gpu_util_pct": None, "error": f"nvidia-smi unavailable: {e}"}

        async def run_one(self, asset: dict) -> GateResult:
            # asset = {"sample_idx": int, "duration_s": int}
            sample = self._sample_nvidia_smi()
            return self.build_result(
                asset_id=f"sample_{asset['sample_idx']:03d}",
                status="ok",
                metrics={
                    **sample,
                    "all_4_models_resident": all(self.substrate._loaded.values()),
                    "sample_idx": asset["sample_idx"],
                },
            )

        async def run_all(self, params: list[dict]) -> list[GateResult]:
            # params = [{"duration_s": int, "sample_interval_s": int}]
            cfg = params[0]
            duration = cfg["duration_s"]
            interval = cfg["sample_interval_s"]
            n_samples = duration // interval
            samples = [{"sample_idx": i, "duration_s": duration} for i in range(n_samples)]
            results: list[GateResult] = []
            for s in samples:
                # Light poke: transcribe a 1-second silent buffer to keep GPU active
                # ... (or skip; depends on substrate readiness)
                r = await self.run_one(s)
                self.emit(r)
                results.append(r)
                await asyncio.sleep(interval)
            return results
    # ... main_async + main + arg parsing (--duration-s default=300, --sample-interval-s default=10)
    ```

    ```python
    """AUDIT-03 engine-swap + Ollama-vs-vLLM overhead measurement runner.

    Two passes:
    1. ENGINE SWAP: render 5 stimuli on Chatterbox, then 5 on Kokoro; measure
       the gap between last-Chatterbox-byte and first-Kokoro-byte as swap-time.
       Validates DR-27 pluggable-TTS architecture viability.
    2. OLLAMA OVERHEAD: same 20 reference probes through both vLLM AWQ-Int4
       and Ollama llama.cpp Q4_K_M; record tokens/sec for each; compute the
       ratio. Grounds the 1.3-1.5x derate scalar (CLAUDE.md §3.1).
    """

    import argparse, asyncio, json, logging, pathlib, subprocess, sys, time
    from gates._runner_base import GateRunner
    from harness.results import GateResult
    from substrate import Substrate
    from substrate.types import VoiceRef

    class AUDIT03Runner(GateRunner):
        def __init__(self, *, substrate: Substrate, ollama_model: str = "qwen3:4b-q4_K_M",
                     **kw):
            super().__init__(substrate=substrate, gate="audit_03", **kw)
            self.ollama_model = ollama_model

        async def _run_swap_pass(self, stimuli_cb: list[dict], stimuli_kk: list[dict]) -> list[GateResult]:
            results = []
            last_cb_t = None
            first_kk_t = None
            # Chatterbox 5
            for s in stimuli_cb:
                t0 = time.perf_counter()
                async for chunk in self.substrate.synthesize(s["text"], VoiceRef(...), engine_hint="chatterbox"):
                    pass
                last_cb_t = time.perf_counter()
                results.append(self.build_result(asset_id=f"cb_{s['stimulus_id']}",
                    status="ok",
                    metrics={"engine":"chatterbox","path":"swap","duration_ms":(last_cb_t-t0)*1000}))
            # Kokoro 5
            for i, s in enumerate(stimuli_kk):
                t0 = time.perf_counter()
                first_chunk_t = None
                async for chunk in self.substrate.synthesize(s["text"], VoiceRef(...), engine_hint="kokoro"):
                    if first_chunk_t is None:
                        first_chunk_t = time.perf_counter()
                end_t = time.perf_counter()
                engine_swap_ms = (first_chunk_t - last_cb_t) * 1000 if i == 0 and first_chunk_t else None
                results.append(self.build_result(asset_id=f"kk_{s['stimulus_id']}",
                    status="ok",
                    metrics={"engine":"kokoro","path":"swap","duration_ms":(end_t-t0)*1000,
                             "engine_swap_ms": engine_swap_ms}))
            return results

        async def _run_ollama_overhead_pass(self, probes: list[dict]) -> list[GateResult]:
            results = []
            for probe in probes:
                # vLLM pass
                t0 = time.perf_counter()
                tokens = 0
                async for chunk in self.substrate.generate(probe["prompt"], grammar=None,
                                                            max_tokens=200):
                    tokens += 1  # rough; refine to count chunk.tokens if available
                vllm_ms = (time.perf_counter() - t0) * 1000
                results.append(self.build_result(asset_id=f"olm_vllm_{probe['probe_id']}",
                    status="ok",
                    metrics={"engine_kind":"vllm","tokens":tokens,"duration_ms":vllm_ms,
                             "tokens_per_sec": tokens/(vllm_ms/1000) if vllm_ms else None}))
                # Ollama pass via subprocess
                try:
                    t0 = time.perf_counter()
                    r = subprocess.run(["ollama","run",self.ollama_model,probe["prompt"]],
                                       capture_output=True, text=True, timeout=120)
                    olm_ms = (time.perf_counter() - t0) * 1000
                    # Parse Ollama's verbose output for eval_count / eval_duration
                    # (run with `--verbose` if needed; tokens/sec derives from those)
                    olm_tokens = len(r.stdout.split())  # rough fallback
                    results.append(self.build_result(asset_id=f"olm_ollama_{probe['probe_id']}",
                        status="ok",
                        metrics={"engine_kind":"ollama","tokens":olm_tokens,
                                 "duration_ms":olm_ms,
                                 "tokens_per_sec": olm_tokens/(olm_ms/1000) if olm_ms else None}))
                except (FileNotFoundError, subprocess.TimeoutExpired) as e:
                    results.append(self.build_result(asset_id=f"olm_ollama_{probe['probe_id']}",
                        status="error",
                        error_kind=type(e).__name__,
                        error_msg=f"ollama_not_installed: {e}"[:500]))
            return results

        async def run_all(self, params: dict) -> list[GateResult]:
            cb = params.get("stimuli_cb", [])
            kk = params.get("stimuli_kk", [])
            probes = params.get("ollama_probes", [])
            r1 = await self._run_swap_pass(cb, kk)
            r2 = await self._run_ollama_overhead_pass(probes)
            for r in r1+r2: self.emit(r)
            return r1 + r2
    # ... main_async + main; load stimuli from assets/tts_pairs/pairs.json[:5] and
    # ... ollama probes from assets/upl_probes/probes.json[:20]
    ```
