    ```python
    """G7 TTS A/B gate runner.

    For each of 30 stimulus text pairs, renders FOUR audio outputs:
    (Chatterbox-cold, Chatterbox-warm, Kokoro-cold, Kokoro-warm). Cold path =
    first render of that stimulus on that engine in this run; warm path =
    second render (cache hit). Listener preference test (SM-72) is a Phase 4
    follow-on; this runner produces the raw audio.
    """

    from __future__ import annotations

    import argparse, asyncio, json, logging, pathlib, sys, time
    from gates._runner_base import GateRunner
    from harness.results import GateResult
    from substrate import Substrate
    from substrate.types import VoiceRef

    logger = logging.getLogger(__name__)

    DEFAULT_VOICE_REFS = {
        "chatterbox": VoiceRef(...),  # cloned voice ref from assets/tts_pairs/reference_clip.wav
        "kokoro": VoiceRef(...),      # neutral voice (e.g., "af_heart" — pick the most-PRD-appropriate)
    }

    class G7Runner(GateRunner):
        def __init__(self, *, substrate, audio_out_dir: pathlib.Path, **kw):
            super().__init__(substrate=substrate, gate="g7", **kw)
            self.audio_out_dir = audio_out_dir
            self.audio_out_dir.mkdir(parents=True, exist_ok=True)

        async def _render_one(self, stimulus: dict, engine: str, path: str) -> GateResult:
            stimulus_id = stimulus["stimulus_id"]
            text = stimulus["text"]
            voice = DEFAULT_VOICE_REFS[engine]
            audio_path = self.audio_out_dir / f"{engine}_{path}_{stimulus_id}.wav"
            t_start = time.perf_counter()
            first_audio_ms: float | None = None
            n_bytes = 0
            try:
                async for chunk in self.substrate.synthesize(text, voice, engine_hint=engine):
                    if first_audio_ms is None:
                        first_audio_ms = (time.perf_counter() - t_start) * 1000
                    n_bytes += len(chunk)
                    # Append-write to file; final file is reconstructed audio
                    with audio_path.open("ab") as f:
                        f.write(chunk)
                total_ms = (time.perf_counter() - t_start) * 1000
            except Exception as e:
                return self.build_result(
                    asset_id=f"{stimulus_id}_{engine}_{path}",
                    status="error",
                    error_kind=type(e).__name__,
                    error_msg=str(e)[:500],
                )
            return self.build_result(
                asset_id=f"{stimulus_id}_{engine}_{path}",
                status="ok",
                tts_first_audio_ms=first_audio_ms,
                metrics={
                    "engine": engine, "path": path, "stimulus_id": stimulus_id,
                    "first_audio_ms": first_audio_ms, "total_audio_ms": total_ms,
                    "audio_path": str(audio_path), "n_bytes": n_bytes,
                },
            )

        async def run_one(self, asset: dict) -> GateResult:
            # G7Runner uses run_all directly (one stimulus = 4 renders); run_one
            # is invoked per render-asset, not per stimulus.
            return await self._render_one(asset["stimulus"], asset["engine"], asset["path"])

        async def run_all(self, stimuli: list[dict]) -> list[GateResult]:
            # Build the 120-render asset list: for each engine, render each stimulus twice
            # (cold = first instance, warm = second).
            render_assets = []
            for engine in ("chatterbox", "kokoro"):
                for stimulus in stimuli:
                    render_assets.append({"stimulus": stimulus, "engine": engine, "path": "cold"})
                for stimulus in stimuli:
                    render_assets.append({"stimulus": stimulus, "engine": engine, "path": "warm"})
            return await super().run_all(render_assets)
    # ... main_async / main + arg parsing mirroring gates/g2/runner.py
    ```
